% call the class
\documentclass[ngerman,article,IEEE,12pt]{docuflow}

% Include definitions (titlepage setup)
\input{definitions.tex}

% === START DOCUMENT ===
\begin{document}

% Roman page numbering (from here on)
\pagenumbering{roman}

% handle abstract, and Kurzfassung (if ngerman)
\DFmakefrontmatter % contains renewed \maketitle

% Arabic page numbering (from here on)
\pagenumbering{arabic}

% Class tutorial
\input{tutorial.tex}
\clearpage

% Example snippets
\input{snippets.tex}
\clearpage

% Font demo, depending on used compile-engine this either shows the 
% LuaLaTeX or the pdfLaTeX font demo
\DFShowFontDemo
\clearpage


% --------------------- Main content -------------------------
\input{content.tex}
\clearpage



% ------------- Reference and Lists content ------------------

% Print bibliography
\printbibliography[title=\bibname, heading=bibintoc]
\clearpage

% Print list of figures
\addcontentsline{toc}{section}{\listfigurename}
\listoffigures
\clearpage

% Print list of tables
\addcontentsline{toc}{section}{\listtablename}
\listoftables
\clearpage

% Print list of codes
\addcontentsline{toc}{section}{\lstlistlistingname}
\lstlistoflistings
\clearpage

% Print acronyms
\addcontentsline{toc}{section}{\acronymname}
\DFabstractSection{\acronymname}
\begin{acronym}[RAM]  % widest acronym for spacing
  \acro{CPU}{Central Processing Unit}
  \acro{RAM}{Random Access Memory}
  \acro{GPU}{Graphics Processing Unit}
\end{acronym}
\clearpage

% Appendix
\appendix
\DFautoSection{\appendixname~A}
\clearpage
\DFautoSection{\appendixname~B}


\end{document}
